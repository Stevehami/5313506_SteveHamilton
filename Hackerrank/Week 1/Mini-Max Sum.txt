void miniMaxSum(int arr_count, int* arr) {
    long total =0;
    long min = LONG_MAX;
    long max = LONG_MIN;
    for (int i=0; i<arr_count;i++) {
        total+=arr[i];
        if (arr[i]>max) {
            max = arr[i];
        }
        if(arr[i]<min){
            min = arr[i];
        }
    }
    long maximum = total - min;
    long minimum = total - max;
    printf("%ld %ld",minimum,maximum);

}
