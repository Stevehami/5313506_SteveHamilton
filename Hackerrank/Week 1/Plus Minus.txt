void plusMinus(int arr_count, int* arr) {
    int pos=0;
    int neg=0;
    int zero=0;
    for(int i=0;i<arr_count;i++){
        if(arr[i]>0){
            pos++;
        }else if (arr[i]<0) {
            neg++;
        }else {
            zero++;
        }
    }
    
    printf("%.6f\n%.6f\n%.6f",(float)pos/arr_count,(float)neg/arr_count,(float)zero/arr_count);

}
